Multiplexing


SOCKET sockets[10] = ...;
fd_set readset;
int result;
do {
  FD_ZERO(&readset);
  for (int i = 0; i < 10; i++) { 
    FD_SET(sockets[i], &readset);
  }
  result = select(0, &readset, nullptr, nullptr, nullptr);
} while (result == -1 && errno == EINTR);
